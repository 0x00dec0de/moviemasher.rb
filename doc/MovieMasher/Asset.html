def self.__download_asset(asset, url, path, job)
  url = asset[:input_url].to_s
  path = url_path(url)
  unless File.exist?(path)
    begin
      FileHelper.safe_path(File.dirname(path))
      asset.download(job: job, path: path)
      raise("zero length #{url}") unless File.size?(path)
    rescue => e
      puts "CAUGHT #{e.is_a?(Error::Job)} #{e.message} #{e.backtrace}"
      raise(Error::JobInput, e.message)
    end
  end
  path
end

def self.__populate_asset_info(asset)
  path = asset[:cached_file]
  case asset[:type]
  when Type::VIDEO
    unless FloatUtil.nonzero(asset[:duration])
      asset[:duration] = Info.get(path, Info::DURATION).to_f
    end
    asset[:no_audio] = !Info.get(path, Info::AUDIO)
    asset[:dimensions] = Info.get(path, Info::DIMENSIONS)
    asset[:no_video] = !asset[:dimensions]
    asset[:av] = Mash.init_av_input(asset)
  when Type::AUDIO
    unless FloatUtil.nonzero(asset[:duration])
      asset[:duration] = Info.get(path, Info::AUDIO_DURATION).to_f
    end
    unless FloatUtil.gtr(asset[:duration], FloatUtil::ZERO)
      asset[:duration] = Info.get(path, Info::VIDEO_DURATION).to_f
    end
  when Type::IMAGE
    asset[:dimensions] = Info.get(path, Info::DIMENSIONS)
    unless asset[:dimensions]
      raise(Error::JobInput, 'could not determine image dimensions')
    end
  end
end

def self.av_type(input)
  case input[:type]
  when Type::AUDIO
    AV::AUDIO_ONLY
  when Type::IMAGE, Type::FONT, Type::FRAME
    AV::VIDEO_ONLY
  when Type::VIDEO, Type::MASH
    if input[:no_audio]
      AV::VIDEO_ONLY
    else
      (input[:no_video] ? AV::AUDIO_ONLY : AV::BOTH)
    end
  end
end

def self.download_asset(asset, job)
  path = nil
  url = asset[:input_url].to_s
  unless url.empty?
    path = __download_asset(asset, url, path, job)
    Info.set(path, Info::AT, Time.now.to_i)
    asset[:cached_file] = path
    __populate_asset_info(asset)
  end
  path
end

def self.url_path(url)
  dir = MovieMasher.configuration[:download_directory]
  dir = MovieMasher.configuration[:render_directory] if dir.to_s.empty?
  hex = Digest::SHA2.new(256).hexdigest(url)
  Path.concat(dir, "#{hex}/#{Info::DOWNLOADED}#{File.extname(url)}")
end

def __url(type, base_url, u)
  if Type::FILE == type
    Path.concat(base_url, u)
  else
    URI.join(base_url, u).to_s
  end
end

def download_source
  d_source = source
  if source && source.is_a?(Source)
    if source.relative?
      # relative url
      case type
      when Type::THEME, Type::FONT, Type::EFFECT
        d_source = module_source || base_source || source
      else
        d_source = base_source || source
      end
    end
  end
  d_source
end

def preflight(job = nil)
  if source
    self.source = Source.create_if source
    if job
      self[:base_source] = job.base_source unless base_source
      self[:module_source] = job.module_source unless module_source
      # puts "base_source: #{base_source}"
      unless Type::MASH == type && @hash[:mash]
        self[:input_url] = url(base_source, module_source)
      end
    end
  end
  # puts "preflight #{self.class.name} URL: #{self[:input_url]}"
end

    source=(value) click to toggle source

    Describes the download request for the element, as either a URL orHash/Source. If the URL is relative it's based fromJob#base_source. Assets of Type::MASH can point to anything thatresponds with a JSON formatted mash. After download they will passthe parsed Hash to MovieMasher::Mash.new and reset their source to thereturned instance. Alternatively, source can be initially set toa Hash/Mash so as to avoid download.

    String

    A HTTP or HTTPS URL to element, converted to appropriate Source.

    Hash

    Can describe either a download request or, for Type::MASHAssets, a JSON formatted Mash. The former is sent to Source.createwhile the later is sent to MovieMasher::Mash.new.

    Returns

    A Source object or, for Type::MASH Assets afterdownloading, a Mash object.

def url(base_src = nil, module_src = nil)
  u = nil
  if source.is_a?(Source)
    u = source.url
    if source.relative?
      base_src = module_src if module_src && Type::MODULES.include?(type)
      if base_src
        base_url = base_src.url
        base_url = Path.add_slash_end base_url
        u = Path.strip_slash_start(u)
        u = __url(base_src[:type], base_url, u)
      end
    end
  end
  # puts "#{self.class.name}#url URL: #{u}"
  u
end
